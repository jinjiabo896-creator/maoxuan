    let currentMode = 'contradiction';

    function setMode(mode, el) {
        currentMode = mode;

        el.classList.add('active');
    }

    function setStage(stage, el) {
        currentStage = stage;

    }

    function executeStrategy() {
        const input = document.getElementById('userInput').value;
        if(!input) return alert("请先输入战况情报。");

        // 核心修复：确保 ID 正确，且增加逻辑合成
        const resultView = document.getElementById('result-view');
        
        // 逻辑推演引擎
        let thinking = "", quote = "", actions = [];
        const baseData = (STRATEGY_DATABASE[currentMode] && STRATEGY_DATABASE[currentMode][currentStage]) ? 
                         STRATEGY_DATABASE[currentMode][currentStage] : null;

        if(baseData) {
            thinking = baseData.thinking;
            quote = baseData.quote;
            actions = baseData.actions;
        } else {
            // 兜底逻辑推演（当数据库未完全覆盖时）
            thinking = `针对你描述的“${input.substring(0,10)}...”，采用${currentMode}框架分析：当前的困境源于你对“时位”的判断失误。在${currentStage === 'defense' ? '防御' : '相持'}阶段，最忌讳‘乱动’。你要从群众中来，到群众中去，重新审视你的基础。`;
            quote = "“我们要坚持实事求是，不办过头事，不办由于主观盲目性而办不到的事。”";
            actions = ["1. 离开桌面，找 3 个真实客户做深度‘田野调查’。","2. 把你的战术目标由‘收割’改为‘服务’，降低阻力。","3. 重新对齐你的沟通节奏，让它符合客观规律。"];
        }

        document.getElementById('res-thinking').innerText = thinking;
        document.getElementById('res-quote').innerText = quote;
        document.getElementById('res-action').innerHTML = actions.map(a => `<p>• ${a}</p>`).join('');

        resultView.style.display = 'block';
        
        // 存储历史
        saveHistory(input.substring(0,20), currentMode);

        window.scrollTo({top: resultView.offsetTop - 20, behavior: 'smooth'});
    }

    function saveHistory(q, m) {
        let list = JSON.parse(localStorage.getItem('strategy_logs') || '[]');
        list.unshift({ q, m, t: new Date().toLocaleString() });
        list = list.slice(0, 5);
        localStorage.setItem('strategy_logs', JSON.stringify(list));
        showHistory();
    }

    function showHistory() {
        const list = JSON.parse(localStorage.getItem('strategy_logs') || '[]');
        if(list.length === 0) return;
        document.getElementById('history-area').style.display = 'block';
        document.getElementById('history-list').innerHTML = list.map(i => `
            <div class="hist-item"><b>${i.t}</b> | ${i.q}... <br>方案：${i.m}</div>
        `).join('');
    }

    showHistory();
